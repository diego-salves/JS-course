Estruturas Condicionais

if, else if, else, switch, while, do while, for

for (var i = 0; i < 10 ; i++){
console.log(i);
}

var numero = 2;
// declarar sempre antes a variavel
while (numero < 10) {
console.log(numero);
numero++;
}

//do-while, primeiro executa depois checa
var num = 0

do {
console.log(num);
num++;
} while ( num < 10);
